Playbooks

playbooks/precheck.yml

Verifica conectividade SSH e que nginx responde na porta 80 nos 3 hosts.

---
- name: Precheck de conectividade
  hosts: all
  gather_facts: false

  tasks:
    - name: Verificar conectividade SSH (ping Ansible)
      ansible.builtin.ping:

    - name: Verificar que nginx está respondendo na porta 80
      ansible.builtin.uri:
        url: "http://{{ inventory_hostname }}"
        status_code: 200
      delegate_to: localhost

playbooks/security-update.yml

Aplica patches de segurança via yum update --security em todos os hosts.

---
- name: Aplicar patches de segurança
  hosts: all
  gather_facts: false
  become: true

  tasks:
    - name: Aplicar atualizações de segurança via yum
      ansible.builtin.yum:
        name: '*'
        security: true
        state: latest
      register: yum_result

    - name: Exibir resumo das atualizações
      ansible.builtin.debug:
        msg: "Pacotes atualizados: {{ yum_result.changes | default('nenhuma mudança') }}"

playbooks/deploy.yml

Rolling update serial: 1 com block/rescue/always. Instala nginx via role, faz deploy, valida porta 80, e em caso de falha reverte automaticamente.

block — caminho feliz

Executa a role deploy → flush handlers → health check na porta 80 (3 tentativas)

rescue — rollback automático

Restaura index.html.bak → reinicia nginx → loga FAILED_ROLLBACK → chama fail para interromper o rolling

always — auditoria

Registra em /var/log/deploy_audit.log com timestamp, host, env, versão e status

---
- name: Rolling deploy da aplicação web
  hosts: all
  serial: 1
  become: true
  gather_facts: true

  vars:
    app_version: "{{ app_version | default('1.0.0') }}"
    deploy_env: "{{ deploy_env | default('dev') }}"
    audit_log: /var/log/deploy_audit.log

  pre_tasks:
    - name: Garantir que nginx está instalado e configurado
      ansible.builtin.include_role:
        name: webserver

  tasks:
    - name: Deploy com rollback automático em caso de falha
      block:
        - name: Executar deploy do index.html
          ansible.builtin.include_role:
            name: deploy

        - name: Forçar flush dos handlers (reload nginx)
          ansible.builtin.meta: flush_handlers

        - name: Verificar que nginx responde na porta 80 após deploy
          ansible.builtin.uri:
            url: "http://{{ inventory_hostname }}"
            status_code: 200
          delegate_to: localhost
          retries: 3
          delay: 5

      rescue:
        - name: Rollback — restaurar index.html anterior
          ansible.builtin.copy:
            src: /usr/share/nginx/html/index.html.bak
            dest: /usr/share/nginx/html/index.html
            remote_src: true

        - name: Registrar falha no log de auditoria
          ansible.builtin.lineinfile:
            path: "{{ audit_log }}"
            line: "{{ ansible_date_time.iso8601 }} | HOST={{ inventory_hostname }} | ENV={{ deploy_env }} | VERSION={{ app_version }} | STATUS=FAILED_ROLLBACK"
            create: true

        - name: Falhar o play para interromper o rolling update
          ansible.builtin.fail:
            msg: "Deploy falhou em {{ inventory_hostname }}. Rollback aplicado."

      always:
        - name: Registrar resultado no log de auditoria
          ansible.builtin.lineinfile:
            path: "{{ audit_log }}"
            line: "{{ ansible_date_time.iso8601 }} | HOST={{ inventory_hostname }} | ENV={{ deploy_env }} | VERSION={{ app_version }} | STATUS=SUCCESS"
            create: true

Roles

roles/webserver/tasks/main.yml

---
- name: Instalar nginx
  ansible.builtin.yum:
    name: nginx
    state: present

- name: Garantir que nginx está habilitado e rodando
  ansible.builtin.service:
    name: nginx
    state: started
    enabled: true

- name: Abrir porta 80 no firewall
  ansible.posix.firewalld:
    service: http
    permanent: true
    state: enabled
    immediate: true
  ignore_errors: true

roles/deploy/tasks/main.yml

---
- name: Fazer backup do index.html atual
  ansible.builtin.copy:
    src: /usr/share/nginx/html/index.html
    dest: /usr/share/nginx/html/index.html.bak
    remote_src: true
  ignore_errors: true

- name: Fazer deploy do index.html
  ansible.builtin.copy:
    content: |
      <!DOCTYPE html>
      <html>
        <head><title>App - {{ deploy_env | upper }}</title></head>
        <body>
          <h1>Ambiente: {{ deploy_env }}</h1>
          <p>Versão: {{ app_version }}</p>
          <p>Host: {{ inventory_hostname }}</p>
          <p>Deploy: {{ ansible_date_time.iso8601 }}</p>
        </body>
      </html>
    dest: /usr/share/nginx/html/index.html
    mode: '0644'
  notify: Reload nginx
